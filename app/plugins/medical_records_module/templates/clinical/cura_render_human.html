{# Render dict/list/leaf values as nested tables and lists — no JSON syntax in the page. #} {%- macro render_human_value(v) -%} {%- if v is none or v is undefined -%} — {%- elif v is mapping -%} {%- if v|length == 0 -%} — {%- else -%} {%- for sk, sv in v.items() %} {%- endfor %}
{{ sk }} {{ render_human_value(sv) }}
{%- endif -%} {%- elif v is string -%} {%- if v | trim | lower == 'unk' -%} Unknown {%- else -%} {{- v -}} {%- endif -%} {%- elif v is sequence -%} {%- if v|length == 0 -%} — {%- else -%}
    {%- for item in v %}
  • {{ render_human_value(item) }}
    • {%- endfor %}
{%- endif -%} {%- else -%} {{- v -}} {%- endif -%} {%- endmacro %}
